\section{Доказательство теоремы Рабина}
\begin{thm}[Рабин]
    Пусть $ n$ нечетное составное число,  $ n > 9$. Тогда  $ S(n) \le \frac{\varphi (n)}{4}$, где $ S(n)$ --- множество свидетелей простоты в тесте Миллера-Рабина.
\end{thm}
